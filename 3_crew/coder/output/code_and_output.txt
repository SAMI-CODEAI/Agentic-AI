# Calculate the first 10,000 terms of the series: 1 - 1/3 + 1/5 - 1/7 + ...

# Initialize the sum
sum_series = 0.0

# Calculate the sum of the series for the first 10,000 terms
for i in range(10000):
    term = 1 / (2 * i + 1)  # Calculate the reciprocal of the odd numbers
    if i % 2 == 0:
        sum_series += term  # Add the term if the index is even
    else:
        sum_series -= term  # Subtract the term if the index is odd

# Multiply the result by 4
final_result = sum_series * 4

# Print the result
print(final_result)
